var inteiro: ledVermelho, ledVerde, sensorTemp, temperatura;
var booleano: alarmeAtivo;

    ledVermelho = 2;
    ledVerde = 3;
    sensorTemp = 4;
    alarmeAtivo = falso;
    
    configurar ledVermelho como saida;
    configurar ledVerde como saida;
    configurar sensorTemp como entrada;
fim

repita
    // LÃª temperatura do sensor
    temperatura = lerAnalogico sensorTemp;
    
    se temperatura > 30 entao
        desligar ledVerde;
        ligar ledVermelho;
        alarmeAtivo = verdadeiro;
    senao
        desligar ledVermelho;
        ligar ledVerde;
        alarmeAtivo = falso;
    fim
    
    esperar 1000;
fim
